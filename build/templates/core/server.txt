const express = require('express');
const requireRoutes = require('./helper');
const app = express();
 

app.use(express.json()); 

requireRoutes(app, [
    { dir: "routes/api", prefix: "/api" },
    { dir: "routes/web", prefix: "" },
]);

app.listen(3000, () => {
    console.log('Server running on port 3000');
});
