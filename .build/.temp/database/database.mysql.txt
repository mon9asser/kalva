const mysql = require('mysql2');
const path = require('path');
require('dotenv').config({ path: path.resolve(__dirname, '../../.env') }); 

class Database {

    static instance;
    static dbInstance;

    constructor() {
         
        if (Database.instance) {
            return Database.instance;
        } 

        var settings = {};
        if(process.env.DATABASE_HOST) {
            settings.host = process.env.DATABASE_HOST;
        }
        if(process.env.DATABASE_USER ) {
            settings.user = process.env.DATABASE_USER;
        }
        if(process.env.DATABASE_PASSWORD) {
            settings.password = process.env.DATABASE_PASSWORD;
        }
        if(process.env.DATABASE_NAME) {
            settings.database = process.env.DATABASE_NAME;
        }

        if(process.env.DATABASE_PORT) {
            settings.port = process.env.DATABASE_PORT;
        }  
       
        Database.dbInstance = mysql.createPool(settings);  
        Database.instance = this; 

    }
    
    async is_connected() {
        return new Promise((resolve) => {
            Database.dbInstance.getConnection((err, connection) => {
                if (err) {
                    resolve(false);
                } else {
                    connection.release();
                    resolve(true);
                }
            });
        });
    }

   async query(sql, params = []) {
        try {

            var db = new Database();
            const isConnected = await db.is_connected();
            if (!isConnected) {
                throw new Error("Database connection error");
            }

            const [rows] = await Database.dbInstance
                .promise()
                .query(sql, params);

            return {
                is_error: false,
                message: "success",
                data: rows
            };

        } catch (error) {

            return {
                is_error: true,
                message: error.message,
                data: []
            };

        }
    }
 

}
  

module.exports = { Database };
