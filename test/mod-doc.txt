// Insert 30,000 records for one time

var { Model } = require("./../build/model");

class Transaction extends Model {
    static table = 'transactions';
    static primary_key = 'id';

    static fillable = [
        'order_id',
        'amount',
        'payment_method',
        'status',
        'reference'
    ];
}
 

(async function(){
    const methods = ['card', 'cash', 'wallet'];
    const statuses = ['pending', 'success', 'failed'];

    const CHUNK_SIZE = 1000;
    const TOTAL = 30000;

    for (let i = 0; i < TOTAL; i += CHUNK_SIZE) {
        const batch = [];

        for (let j = 0; j < CHUNK_SIZE && (i + j) < TOTAL; j++) {
            const orderId = Math.floor(Math.random() * 3000) + 1;

            batch.push({
                order_id: orderId,
                amount: Number((Math.random() * 1000 + 5).toFixed(2)),
                payment_method: methods[Math.floor(Math.random() * methods.length)],
                status: statuses[Math.floor(Math.random() * statuses.length)],
                reference: `TXN-${Date.now()}-${i + j}`
            });
        }

        await Transaction.create(batch);
    }
})();
